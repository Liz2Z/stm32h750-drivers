```rust
#![no_std]
#![no_main]

use panic_halt as _;
use stm32h7::stm32h750v::Peripherals;

#[cortex_m_rt::entry]
fn main() -> ! {
    let dp = Peripherals::take().unwrap();

    // 启用 GPIO 时钟
    dp.RCC.ahb4enr.modify(|_, w| {
        w.gpioaen().enabled()
         .gpioben().enabled()
         .gpiocen().enabled()
         .gpioden().enabled()
         .gpioeen().enabled()
    });

    // 配置 SPI 引脚 (PB12-15) 和复位引脚 (PE0)
    let gpiob = dp.GPIOB;
    let gpioe = dp.GPIOE;

    // SPI_SCK (PB13), SPI_MISO (PB14), SPI_MOSI (PB15) - 复用功能
    gpiob.moder.modify(|_, w| {
        w.moder13().alternate();
        w.moder14().alternate();
        w.moder15().alternate();
        w.moder12().output()  // NSS 作为普通输出
    });

    // 设置复用功能 AF5 (SPI2)
    gpiob.afrh.modify(|_, w| {
        w.afrh13().af5();
        w.afrh14().af5();
        w.afrh15().af5()
    });

    // RC522_RST (PE0) - 推挽输出
    gpioe.moder.modify(|_, w| w.moder0().output());
    gpioe.otyper.modify(|_, w| w.ot0().push_pull());
    gpioe.bsrr.write(|w| w.bs0().set());

    loop {}
}
```

```rust
// src/spi_hal.rs

use stm32h7::stm32h750v::{SPI2, GPIOB};
use core::ops::Deref;

pub struct SpiHal {
    spi: SPI2,
    nss: NssControl,
}

struct NssControl {
    gpio: GPIOB,
}

impl SpiHal {
    pub fn new(spi: SPI2, gpiob: GPIOB) -> Self {
        // 启用 SPI2 时钟
        unsafe {
            let rcc = &*stm32h7::stm32h750v::RCC::ptr();
            rcc.apb1lenr.modify(|_, w| w.spi2en().enabled());
        }

        // 配置 SPI: Mode 0, 8-bit, Master
        spi.cfg1.write(|w| {
            w.mbr().div4()      // PCLK / 4 (< 1MHz for init)
             .dsize().bits8()   // 8 data bits
             .fthlv().one_frame()
        });

        spi.cfg2.write(|w| {
            w.cpha().first_edge()    // CPHA = 0
             .cpol().idle_low()      // CPOL = 0
             .master().master()      // Master mode
             .ssel().software()      // Software NSS
        });

        spi.cr1.write(|w| w.ssi().set());  // Internal NSS high

        // 启用 SPI
        spi.cr1.modify(|_, w| w.spe().enabled());

        Self {
            spi,
            nss: NssControl { gpio: gpiob },
        }
    }

    fn cs_low(&mut self) {
        self.nss.gpio.bsrr.write(|w| w.br12().reset());
    }

    fn cs_high(&mut self) {
        self.nss.gpio.bsrr.write(|w| w.bs12().set());
    }

    pub fn transfer(&mut self, data: u8) -> nb::Result<u8, ()> {
        // 等待 TXE
        if self.spi.sr.read().txp().is_not_empty() {
            return Err(nb::Error::WouldBlock);
        }

        // 发送数据
        self.spi.txdr.write(|w| unsafe { w.txdata().bits(data) });

        // 等待 RXP
        if self.spi.sr.read().rxp().is_empty() {
            return Err(nb::Error::WouldBlock);
        }

        // 读取数据
        Ok(self.spi.rxdr.read().rxdata().bits())
    }

    pub fn write(&mut self, data: &[u8]) -> Result<(), ()> {
        for &byte in data {
            nb::block!(self.transfer(byte))?;
        }
        Ok(())
    }

    pub fn read(&mut self, buf: &mut [u8]) -> Result<(), ()> {
        for byte in buf.iter_mut() {
            *byte = nb::block!(self.transfer(0xFF))?;
        }
        Ok(())
    }
}
```

### Step 3: 实现 RC522 基础结构和初始化

```rust
// src/rc522.rs (继续)

use crate::types::{CardType, KeyType, Error};
use embedded_hal::digital::OutputPin;

pub struct RC522<SPI, RST> {
    spi: SPI,
    rst: RST,
}

impl<SPI, E, RST> RC522<SPI, RST>
where
    SPI: embedded_hal::spi::SpiBus<Error = E>,
    RST: OutputPin<Error = E>,
{
    pub fn new(mut spi: SPI, mut rst: RST) -> Result<Self, Error<E>> {
        // 硬件复位
        rst.set_low().map_err(Error::Spi)?;
        // 延时
        for _ in 0..1000 { cortex_m::asm::nop(); }
        rst.set_high().map_err(Error::Spi)?;
        for _ in 0..10000 { cortex_m::asm::nop(); }

        let mut rc522 = Self { spi, rst };

        // 软复位
        rc522.write_reg(COMMAND_REG, CMD_SOFT_RESET)?;

        // 配置: CRC中断、定时器中断、自动重冲
        rc522.write_reg(COM_I_EN_REG, 0x7F)?;
        rc522.write_reg(DIV_IRQ_REG, 0x00)?;

        // 开启天线
        rc522.set_antenna(true)?;

        Ok(rc522)
    }

    fn read_reg(&mut self, reg: u8) -> Result<u8, Error<E>> {
        // RC522 读取: 0x03 | (reg << 1)
        let addr = 0x80 | reg;
        let mut tx = [addr, 0x00];
        let mut rx = [0u8; 2];

        self.spi.transfer(&mut rx, &tx).map_err(Error::Spi)?;

        Ok(rx[1])
    }

    fn write_reg(&mut self, reg: u8, val: u8) -> Result<(), Error<E>> {
        // RC522 写入: 0x02 | (reg << 1)
        let addr = reg & 0x7F;
        let tx = [addr, val];
        let mut rx = [0u8; 2];

        self.spi.transfer(&mut rx, &tx).map_err(Error::Spi)?;

        Ok(())
    }

    fn set_antenna(&mut self, on: bool) -> Result<(), Error<E>> {
        let val = if on { 0x03 } else { 0x00 };
        self.write_reg(0x26, val)
    }
}
```

**Files:**

### Step 1: 实现寻卡功能

```rust
impl<SPI, E, RST> RC522<SPI, RST>
where
    SPI: embedded_hal::spi::SpiBus<Error = E>,
    RST: OutputPin<Error = E>,
{
    /// 寻卡
    pub fn request(&mut self) -> Result<CardType, Error<E>> {
        // 配置 BitFraming
        self.write_reg(BIT_FRAMING_REG, 0x07)?;

        // 发送 REQALL 命令
        let cmd = PICC_REQALL;
        self.write_reg(COMMAND_REG, CMD_IDLE)?;
        self.write_reg(FIFO_DATA_REG, cmd)?;
        self.write_reg(COMMAND_REG, CMD_TRANSCEIVE)?;

        // 配置开始发送
        self.write_reg(BIT_FRAMING_REG, 0x80 | 0x07)?;

        // 等待完成
        let mut timeout = 1000;
        while self.read_reg(COM_IRQ_REG)? & 0x01 == 0 {
            timeout -= 1;
            if timeout == 0 {
                return Err(Error::Timeout);
            }
        }

        // 检查错误
        let status = self.read_reg(ERROR_REG)?;
        if status & 0x08 != 0 {
            return Err(Error::Collision);
        }

        // 读取返回数据
        let n = self.read_reg(FIFOLEVEL_REG)?;
        if n != 2 {
            return Err(Error::NoCard);
        }

        let byte1 = self.read_reg(FIFO_DATA_REG)?;
        let byte2 = self.read_reg(FIFO_DATA_REG)?;

        match (byte1, byte2) {
            (0x02, 0x00) => Ok(CardType::Mifare1K),
            (0x04, 0x00) => Ok(CardType::MifareUltralight),
            (0x02, 0x04) => Ok(CardType::Mifare4K),
            _ => Ok(CardType::Unknown),
        }
    }
}
```

```rust
impl<SPI, E, RST> RC522<SPI, RST>
where
    SPI: embedded_hal::spi::SpiBus<Error = E>,
    RST: OutputPin<Error = E>,
{
    /// 防冲突检测，获取 UID
    pub fn anticoll(&mut self) -> Result<[u8; 4], Error<E>> {
        let mut uid = [0u8; 4];

        // 配置位帧
        self.write_reg(BIT_FRAMING_REG, 0x00)?;

        // 清空 FIFO
        self.write_reg(FIFOLEVEL_REG, 0x80)?;

        // 写入防冲突命令
        self.write_reg(FIFO_DATA_REG, PICC_ANTICOLL1)?;
        self.write_reg(FIFO_DATA_REG, 0x20)?;

        // 执行
        self.write_reg(COMMAND_REG, CMD_TRANSCEIVE)?;
        self.write_reg(BIT_FRAMING_REG, 0x80)?;

        // 等待完成
        let mut timeout = 1000;
        while self.read_reg(COM_IRQ_REG)? & 0x01 == 0 {
            timeout -= 1;
            if timeout == 0 {
                return Err(Error::Timeout);
            }
        }

        // 读取 UID
        let n = self.read_reg(FIFOLEVEL_REG)?;
        if n != 5 {
            return Err(Error::Collision);
        }

        for i in 0..5 {
            let byte = self.read_reg(FIFO_DATA_REG)?;
            if i < 4 {
                uid[i] = byte;
            }
        }

        Ok(uid)
    }
}
```

```rust
impl<SPI, E, RST> RC522<SPI, RST>
where
    SPI: embedded_hal::spi::SpiBus<Error = E>,
    RST: OutputPin<Error = E>,
{
    /// 认证
    pub fn authenticate(
        &mut self,
        block: u8,
        key_type: KeyType,
        key: &[u8; 6],
        uid: &[u8; 4],
    ) -> Result<(), Error<E>> {
        let cmd = match key_type {
            KeyType::KeyA => PICC_AUTHENT1A,
            KeyType::KeyB => PICC_AUTHENT1B,
        };

        // 清空 FIFO
        self.write_reg(FIFOLEVEL_REG, 0x80)?;

        // 写入认证数据
        self.write_reg(FIFO_DATA_REG, cmd)?;
        self.write_reg(FIFO_DATA_REG, block)?;
        for &b in key.iter() {
            self.write_reg(FIFO_DATA_REG, b)?;
        }
        for &b in uid.iter() {
            self.write_reg(FIFO_DATA_REG, b)?;
        }

        // 执行认证
        self.write_reg(COMMAND_REG, CMD_AUTHENT)?;
        self.write_reg(BIT_FRAMING_REG, 0x00)?;

        // 等待完成
        let mut timeout = 5000;
        while self.read_reg(COM_IRQ_REG)? & 0x01 == 0 {
            timeout -= 1;
            if timeout == 0 {
                return Err(Error::Timeout);
            }
        }

        // 检查状态
        let status = self.read_reg(STATUS_REG)?;
        if status & 0x08 != 0 {
            return Err(Error::Timeout);  // 认证失败
        }

        Ok(())
    }
}
```

```rust
    /// 读取一个数据块 (16 字节)
    pub fn read(&mut self, block: u8) -> Result<[u8; 16], Error<E>> {
        let mut data = [0u8; 16];

        // 清空 FIFO
        self.write_reg(FIFOLEVEL_REG, 0x80)?;

        // 写入读命令
        self.write_reg(COMMAND_REG, CMD_IDLE)?;
        self.write_reg(FIFO_DATA_REG, PICC_READ)?;
        self.write_reg(FIFO_DATA_REG, block)?;

        // 执行
        self.write_reg(COMMAND_REG, CMD_TRANSCEIVE)?;
        self.write_reg(BIT_FRAMING_REG, 0x80)?;

        // 等待完成
        let mut timeout = 5000;
        while self.read_reg(COM_IRQ_REG)? & 0x01 == 0 {
            timeout -= 1;
            if timeout == 0 {
                return Err(Error::Timeout);
            }
        }

        // 检查错误
        let error = self.read_reg(ERROR_REG)?;
        if error & 0x08 != 0 {
            return Err(Error::Collision);

        // 读取数据 (16 字节 + 2 字节 CRC)
        let n = self.read_reg(FIFOLEVEL_REG)?;
        if n < 18 {
            return Err(Error::NoCard);
        }

        for i in 0..16 {
            data[i] = self.read_reg(FIFO_DATA_REG)?;
        }

        Ok(data)
    }
```

```rust
where
    SPI: embedded_hal::spi::SpiBus<Error = E>,
    RST: OutputPin<Error = E>,
{
    /// 写入一个数据块 (16 字节)
    pub fn write(&mut self, block: u8, data: &[u8; 16]) -> Result<(), Error<E>> {
        // 清空 FIFO
        self.write_reg(FIFOLEVEL_REG, 0x80)?;

        // 写入写命令
        self.write_reg(COMMAND_REG, CMD_IDLE)?;
        self.write_reg(FIFO_DATA_REG, PICC_WRITE)?;
        self.write_reg(FIFO_DATA_REG, block)?;

        // 执行
        self.write_reg(COMMAND_REG, CMD_TRANSCEIVE)?;
        self.write_reg(BIT_FRAMING_REG, 0x80)?;

        // 等待完成
        let mut timeout = 5000;
        while self.read_reg(COM_IRQ_REG)? & 0x01 == 0 {
            timeout -= 1;
            if timeout == 0 {
                return Err(Error::Timeout);
            }
        }

        // 检查是否需要发送数据
        let n = self.read_reg(FIFOLEVEL_REG)?;
        if n != 1 {
            return Err(Error::Collision);

        let byte = self.read_reg(FIFO_DATA_REG)?;
```

```rust
#![no_std]
#![no_main]

use panic_halt as _;
use stm32h7::stm32h750v::Peripherals;
use cortex_m_rt::entry;
use cortex_m::delay::Delay;
use stm32h7::stm32h750v::CorePeripherals;

mod spi_hal;
mod rc522;
mod types;

use spi_hal::SpiHal;
use rc522::RC522;
use types::CardType;

const DEFAULT_KEY: [u8; 6] = [0xFFu8; 6];

#[entry]
fn main() -> ! {
    let cp = CorePeripherals::take().unwrap();
    let dp = Peripherals::take().unwrap();

    let mut delay = Delay::new(cp.SYST, 120_000_000);  // 120MHz

    // 启用 GPIO 时钟
    dp.RCC.ahb4enr.modify(|_, w| {
        w.gpioaen().enabled()
         .gpioben().enabled()
         .gpiocen().enabled()
         .gpioden().enabled()
         .gpioeen().enabled()
    });

    // 配置 SPI 引脚
    let gpiob = dp.GPIOB;
    let gpioe = dp.GPIOE;

    gpiob.moder.modify(|_, w| {
        w.moder13().alternate();
        w.moder14().alternate();
        w.moder15().alternate();
        w.moder12().output()
    });

    gpiob.afrh.modify(|_, w| {
        w.afrh13().af5();
        w.afrh14().af5();
        w.afrh15().af5()
    });

    gpioe.moder.modify(|_, w| w.moder0().output());
    gpioe.otyper.modify(|_, w| w.ot0().push_pull());
    gpioe.bsrr.write(|w| w.bs0().set());

    // 初始化 SPI 和 RC522
    let spi = SpiHal::new(dp.SPI2, dp.GPIOB);
    let mut rc522 = RC522::new(spi, gpioe).unwrap();

    loop {
        // 寻卡
        match rc522.request() {
            Ok(CardType::Mifare1K) => {
                // 获取 UID
                if let Ok(uid) = rc522.anticoll() {
                    // 尝试读取块 4
                    if rc522.authenticate(4, types::KeyType::KeyA, &DEFAULT_KEY, &uid).is_ok() {
                        if let Ok(data) = rc522.read(4) {
                            // 成功读取，data 包含 16 字节数据
                            // 可以在这里添加处理逻辑
                        }
                    }
                }
            }
            _ => {
                // 无卡或其他类型
            }
        }

        delay.delay_ms(500u32);
    }
}
```

### Step 1: 添加选择卡片功能

```rust
impl<SPI, E, RST> RC522<SPI, RST>
where
    SPI: embedded_hal::spi::SpiBus<Error = E>,
    RST: OutputPin<Error = E>,
{
    /// 选择卡片
    pub fn select_card(&mut self, uid: &[u8; 4]) -> Result<(), Error<E>> {
        // 清空 FIFO
        self.write_reg(FIFOLEVEL_REG, 0x80)?;

        // 发送选择命令
        self.write_reg(COMMAND_REG, CMD_IDLE)?;
        self.write_reg(FIFO_DATA_REG, 0x93)?;  // SEL
        self.write_reg(FIFO_DATA_REG, 0x70)?;  // NVB (7 bytes)

        for &b in uid.iter() {
            self.write_reg(FIFO_DATA_REG, b)?;
        }

        // 计算并添加 CRC (简化，使用 0x00)
        self.write_reg(FIFO_DATA_REG, 0x00)?;
        self.write_reg(FIFO_DATA_REG, 0x00)?;

        // 执行
        self.write_reg(COMMAND_REG, CMD_TRANSCEIVE)?;
        self.write_reg(BIT_FRAMING_REG, 0x80)?;

        // 等待完成
        let mut timeout = 5000;
        while self.read_reg(COM_IRQ_REG)? & 0x01 == 0 {
            timeout -= 1;
            if timeout == 0 {
                return Err(Error::Timeout);
            }
        }

        // 读取响应
        let n = self.read_reg(FIFOLEVEL_REG)?;
        if n < 2 {
            return Err(Error::NoCard);
        }

        let byte1 = self.read_reg(FIFO_DATA_REG)?;
        let byte2 = self.read_reg(FIFO_DATA_REG)?;

        if byte1 != uid[0] {
            return Err(Error::Collision);
        }

        Ok(())
    }
}
```
